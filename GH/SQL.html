전체 데이터베이스 테이블 생성 및 RLS 정책 일괄 적용

Supabase 데이터베이스에서 처음으로 테이블들을 구축할 때 실행하는 단일 스크립트입니다. 본 스크립트는 아래의 작업을 수행합니다.

  • master_project 테이블 생성: 연도(2025~2030), 반기(상반기/하반기), 시나리오 구분을 가지는 마스터 프로젝트 테이블
  • 하위 4대 실적 테이블 생성 (land_costs, development_costs, supplies, recoveries): 마스터 프로젝트 테이블의 외래키 복합키 참조(ON UPDATE/DELETE CASCADE 적용)
  • 보안 제약 설정 (RLS): 모든 테이블에 Row Level Security를 적용하고, 오직 Supabase Auth 로그인 세션(authenticated)을 가진 인증 유저만 조작을 허용하도록 설정
실행 안내: Supabase 프로젝트 대시보드의 SQL Editor 메뉴에서 새 쿼리창(New Query)을 열고, 아래의 SQL 코드를 붙여넣어 실행(Run)해 주십시오. 기존 테이블이 존재하는 상태에서 초기화 후 재구축하려면 DROP TABLE IF EXISTS ... CASCADE;를 스크립트 최상단에 추가하여 실행해 주시면 됩니다.
postgresql 
-- =========================================================================
-- 1. 마스터 프로젝트 테이블 생성 (master_project)
-- =========================================================================
CREATE TABLE master_project (
    year INT NOT NULL,
    half VARCHAR(10) NOT NULL,
    scenario VARCHAR(50) NOT NULL,
    description TEXT,
    PRIMARY KEY (year, half, scenario),
    CONSTRAINT chk_project_year CHECK (year BETWEEN 2025 AND 2030),
    CONSTRAINT chk_project_half CHECK (half IN ('상반기', '하반기'))
);

-- =========================================================================
-- 2. 하위 4대 실적 및 예산 테이블 생성 (CASCADE 외래키 적용)
-- =========================================================================

-- 용지비 (land_costs)
CREATE TABLE land_costs (
    id SERIAL PRIMARY KEY,
    year INT NOT NULL,
    half VARCHAR(10) NOT NULL,
    scenario VARCHAR(50) NOT NULL,
    detail_item VARCHAR(200) NOT NULL,
    target_year INT NOT NULL,
    amount NUMERIC DEFAULT 0,
    FOREIGN KEY (year, half, scenario) REFERENCES master_project(year, half, scenario) ON DELETE CASCADE ON UPDATE CASCADE
);

-- 조성비 (development_costs)
CREATE TABLE development_costs (
    id SERIAL PRIMARY KEY,
    year INT NOT NULL,
    half VARCHAR(10) NOT NULL,
    scenario VARCHAR(50) NOT NULL,
    detail_item VARCHAR(200) NOT NULL,
    target_year INT NOT NULL,
    amount NUMERIC DEFAULT 0,
    FOREIGN KEY (year, half, scenario) REFERENCES master_project(year, half, scenario) ON DELETE CASCADE ON UPDATE CASCADE
);

-- 분양 수입 (supplies)
CREATE TABLE supplies (
    id SERIAL PRIMARY KEY,
    year INT NOT NULL,
    half VARCHAR(10) NOT NULL,
    scenario VARCHAR(50) NOT NULL,
    detail_item VARCHAR(200) NOT NULL,
    target_year INT NOT NULL,
    amount NUMERIC DEFAULT 0,
    FOREIGN KEY (year, half, scenario) REFERENCES master_project(year, half, scenario) ON DELETE CASCADE ON UPDATE CASCADE
);

-- 자금 회수액 (recoveries)
CREATE TABLE recoveries (
    id SERIAL PRIMARY KEY,
    year INT NOT NULL,
    half VARCHAR(10) NOT NULL,
    scenario VARCHAR(50) NOT NULL,
    detail_item VARCHAR(200) NOT NULL,
    target_year INT NOT NULL,
    amount NUMERIC DEFAULT 0,
    FOREIGN KEY (year, half, scenario) REFERENCES master_project(year, half, scenario) ON DELETE CASCADE ON UPDATE CASCADE
);

-- =========================================================================
-- 3. Row Level Security (RLS) 활성화
-- =========================================================================
ALTER TABLE master_project ENABLE ROW LEVEL SECURITY;
ALTER TABLE land_costs ENABLE ROW LEVEL SECURITY;
ALTER TABLE development_costs ENABLE ROW LEVEL SECURITY;
ALTER TABLE supplies ENABLE ROW LEVEL SECURITY;
ALTER TABLE recoveries ENABLE ROW LEVEL SECURITY;

-- =========================================================================
-- 4. RLS 정책 생성 (인증된 로그인 사용자에게만 허용)
-- =========================================================================
DROP POLICY IF EXISTS "Allow authenticated access to master_project" ON master_project;
CREATE POLICY "Allow authenticated access to master_project" ON master_project
    FOR ALL TO authenticated USING (true) WITH CHECK (true);

DROP POLICY IF EXISTS "Allow authenticated access to land_costs" ON land_costs;
CREATE POLICY "Allow authenticated access to land_costs" ON land_costs
    FOR ALL TO authenticated USING (true) WITH CHECK (true);

DROP POLICY IF EXISTS "Allow authenticated access to development_costs" ON development_costs;
CREATE POLICY "Allow authenticated access to development_costs" ON development_costs
    FOR ALL TO authenticated USING (true) WITH CHECK (true);

DROP POLICY IF EXISTS "Allow authenticated access to supplies" ON supplies;
CREATE POLICY "Allow authenticated access to supplies" ON supplies
    FOR ALL TO authenticated USING (true) WITH CHECK (true);

DROP POLICY IF EXISTS "Allow authenticated access to recoveries" ON recoveries;
CREATE POLICY "Allow authenticated access to recoveries" ON recoveries
    FOR ALL TO authenticated USING (true) WITH CHECK (true);
